116static int NextPowerOfTwo(int n) {
117
118 if (NGIsPowerOfTwo(n)) return n;
119
120 for (int i = kMaxTextureSizeExp - 1; i > 0; i--) {
121
122 if (n & (1 << i)) return (1 << (i+1));
123
124 }
125
126 return kMaxTextureSize(1 << kMaxTextureSizeExp);
