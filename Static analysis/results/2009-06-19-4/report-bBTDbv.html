60static inline NGTextureFormat GetImageFormat(CGImageRef image) {
61
62 CGImageAlphaInfo alpha = CGImageGetAlphaInfo(image);
63 bool_Bool hasAlpha = (alpha != kCGImageAlphaNone && alpha != kCGImageAlphaNoneSkipLast && alpha != kCGImageAlphaNoneSkipFirst);
64
65// CGBitmapInfo info = CGImageGetBitmapInfo(image); //! ATH TEST
66
67 CGColorSpaceRef color = CGImageGetColorSpace(image);
68
69 if (color != NULL((void*)0)) {
70
71 if (CGColorSpaceGetModel(color) == kCGColorSpaceModelMonochrome)
72 return (hasAlpha? NGTextureFormat_LA_88 : NGTextureFormat_A_8);
73
74 int bpp = CGImageGetBitsPerPixel(image);
75
76 if (bpp == 16) return (hasAlpha? NGTextureFormat_RGBA_5551 : NGTextureFormat_RGB_565);
77
78 return (hasAlpha? NGTextureFormat_RGBA_8888 : NGTextureFormat_RGB_888);
79 }
80
81 return NGTextureFormat_A_8;
82}

145#define kMaxTextureSize(1 << kMaxTextureSizeExp) (1 << kMaxTextureSizeExp)

147static inline int NextPowerOfTwo(int n)
148{
149 if (NGIsPowerOfTwo(n))
150 return n;
151
152 for (int i = kMaxTextureSizeExp - 1; i > 0; i--)
153 {
154 if (n & (1 << i))
155 return (1 << (i+1));
156 }
157
158 return kMaxTextureSize(1 << kMaxTextureSizeExp);
159}
